\section{Room temperature control formal specification}

\noindent The EFSM of the room temperature control is the tuple $S = (Q, \Sigma_1, \Sigma_2, q_0, V, \Lambda)$, where
\noindent $Q = \{idle, warm up, configuration\}$\\
\noindent $\Sigma_1 = \{interrupt, after(3 min), after(2 s), cancel\}$\\
\noindent $\Sigma_2 = \{fan~on, fan~off, furnace~on, furnace~off, prolonged~beep~sound\}$\\
\noindent $q_0: idle$\\
\noindent $V: \{C, T, D, T.F\}$\\
\noindent $\Lambda$: Transition specifications\\
\indent 1. $\rightarrow idle$\\
\indent 2. $idle \xrightarrow {\text { shut off/ (fan off ; furnace off)}} off$\\
\indent 3. $idle \xrightarrow {\text { [C $\geq$ D]}} idle$\\
\indent 4. $idle \xrightarrow {\text { [C $\leq$ D-1] / (fan off ; furnace on)}} warm~up$\\
\indent 5. $warm~up \xrightarrow {\text { [T.F $<$ D+1] }} warm~up$\\
\indent 6. $warm~up \xrightarrow {\text { [T.F $\geq$ D+1] }} idle$\\
\indent 7. $warm~up \xrightarrow {\text { interrupt/furnace off }} configuration$\\

\noindent The EFSM of the configuration is the tuple $S = (Q, \Sigma_1, \Sigma_2, q_0, V, \Lambda)$, where\\
\noindent $Q = \{input, add, override, exit\}$\\
\noindent $\Sigma_1 = \{after(1 min)\}$\\
% \noindent $\Sigma_2 = \{, unlock, beep\}$\\
\noindent $q_0: input$\\
\noindent $V: triplet$\\
\noindent $\Lambda$: Transition specifications\\
\indent 1. $\rightarrow input$\\
\indent 2. $input \xrightarrow {\text { [triplet does not exist]}} add$\\
\indent 3. $input \xrightarrow {\text { [triplet exist]}} override$\\
\indent 4. $input \xrightarrow {\text { after 1 inactive min}} exit$\\
\indent 5. $add \xrightarrow {\text { [else]}} input$\\
\indent 6. $add \xrightarrow {\text { [completed]}} exit$\\
\indent 7. $override \xrightarrow {\text { [else]}} input$\\
\indent 8. $override \xrightarrow {\text { [completed]}} exit$\\
